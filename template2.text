<script>
  // Toggle function for paragraph-by-paragraph switch
  function toggleParagraph(p) {
    const high = p.querySelector('.high');
    const low = p.querySelector('.low');
    if(high.style.display === 'none') {
      high.style.display = 'inline';
      low.style.display = 'none';
    } else {
      high.style.display = 'none';
      low.style.display = 'inline';
    }
  }
</script>
